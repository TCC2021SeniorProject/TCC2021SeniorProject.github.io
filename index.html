Start async def Start(self):
  if  device_on == true:
    await template().Charge()

Charge async def Charge(self):
  if  fully_charged == true:
    await template().Run()

Run async def Run(self):
  if  mode == 0:
    await template().Start()
  if  mode == 1:
    await template().Sync()

Sync async def Sync(self):
  if  mode == 2:
    await template().Run()
  if  mode == 3:
    await template().Check_Status()

Check Status async def Check_Status(self):
  if  mode == 2:
    await template().Run()
  if  mode == 4:
    await template().Return()
